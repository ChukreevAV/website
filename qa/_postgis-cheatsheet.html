Некоторые команды для работы с БД PostgreSQL/PostGIS

Аннотация

CREATE DATABASE vmap0 OWNER postgres;

 

sudo -u pgsql createlang plpgsql vmap0


sudo psql -U pgsql -d vmap0 -f /usr/share/pgsql/contrib/lwpostgis.sql
sudo psql -U pgsql -d vmap0 -f /usr/local/share/postgis/spatial_ref_sys.sql

psql -U labinfo -d vmap0 -f veg_tree_a.sql

 for map in *sql; do psql -U labinfo -d vmap0 -f $map; done

 

psql -U pgsql vmap0 labinfo

\i /path/to/file.sql

GRANT SELECT ON TABLE veg_tree_a TO gis1;
